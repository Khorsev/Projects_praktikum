    book.title,
    SUM(buy_book.amount) AS Количество,
    SUM(book.price) AS Сумма
FROM

    LEFT JOIN buy_step
    USING(buy_id)

    date_step_end IS NOT NULL AND step_id = 1
GROUP BY
